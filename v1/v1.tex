\item What is the output of following program?
\begin{lstlisting}[language=C++]
#include <iostream>
using namespace std;
int main(){
    int x = 7, y = 2, cnt = -1;
    while(x > y){
        if(x % y == 0){
            cnt++;
        }
        x--;
    }
    cout << cnt;
    return 0;
}
\end{lstlisting}
\begin{enumerate}
    \item 2+
    \item 1
    \item 3
    \item None of them
\end{enumerate}
